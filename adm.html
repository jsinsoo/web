제시 코드

class Polynomial:
    def __init__(self, coefficients):
        self.coefficients = coefficients

    def value(self, x):
        total = 0
        for power, coef in enumerate(self.coefficients):
            total += coef * (x ** power)
        return total

    def derivative(self):
        new_coefficients = []
        for power, coef in enumerate(self.coefficients[1:], start=1):
            new_coefficients.append(power * coef)
        return Polynomial(new_coefficients)

    def __str__(self):
        terms = []
        for power, coef in enumerate(self.coefficients):
            terms.append(f"{coef}x^{power}")
        return " + ".join(terms)

p = Polynomial([2, -3, 1])   # 2 - 3x + x^2
dp = p.derivative()

print("p(x) =", p)
print("p(5) =", p.value(5))
print("p'(x) =", dp)
print("p'(5) =", dp.value(5))

class NumericalDerivative:
    def __init__(self, function, h=1e-5):
        self.function = function
        self.h = h

    def derivative_at(self, x):
        f = self.function
        h = self.h
        return (f(x + h) - f(x - h)) / (2 * h)

    def tangent_line(self, x):
        slope = self.derivative_at(x)
        y = self.function(x)
        return slope, y - slope * x

def f(x):
    return x**3 - 2*x + 1

D = NumericalDerivative(f)

print("f'(2) 근사:", D.derivative_at(2))
m, b = D.tangent_line(2)
print("접선: y =", m, "x +", b)

class NumericalIntegral:
    def __init__(self, function, a, b, n=1000):
        self.function = function
        self.a = a
        self.b = b
        self.n = n

    def trapezoid(self):
        dx = (self.b - self.a) / self.n
        total = 0

        for i in range(self.n):
            x1 = self.a + i * dx
            x2 = x1 + dx
            total += (self.function(x1) + self.function(x2)) * dx / 2

        return total

def f(x):
    return x**2

I = NumericalIntegral(f, 0, 1, n=10000)

print("적분 근사값:", I.trapezoid())

class Vector:
    def __init__(self, components):
        self.components = components

    def dot(self, other):
        return sum(a * b for a, b in zip(self.components, other.components))

    def magnitude(self):
        return sum(x**2 for x in self.components) ** 0.5

    def cosine_similarity(self, other):
        return self.dot(other) / (self.magnitude() * other.magnitude())

    def add(self, other):
        return Vector([a + b for a, b in zip(self.components, other.components)])

v1 = Vector([1, 2, 3])
v2 = Vector([4, 5, 6])

print("내적:", v1.dot(v2))
print("v1 크기:", v1.magnitude())
print("코사인 유사도:", v1.cosine_similarity(v2))
print("벡터합:", v1.add(v2).components)

class Matrix:
    def __init__(self, data):
        self.data = data
        self.rows = len(data)
        self.cols = len(data[0])

    def transpose(self):
        result = []
        for j in range(self.cols):
            row = []
            for i in range(self.rows):
                row.append(self.data[i][j])
            result.append(row)
        return Matrix(result)

    def multiply(self, other):
        result = []
        for i in range(self.rows):
            row = []
            for j in range(other.cols):
                value = 0
                for k in range(self.cols):
                    value += self.data[i][k] * other.data[k][j]
                row.append(value)
            result.append(row)
        return Matrix(result)

A = Matrix([[1, 2], [3, 4]])
B = Matrix([[5, 6], [7, 8]])

print("A의 전치:", A.transpose().data)
print("AB:", A.multiply(B).data)

import math

class LinearTransform2D:
    def __init__(self, matrix):
        self.matrix = matrix

    def apply(self, vector):
        x, y = vector
        a, b = self.matrix[0]
        c, d = self.matrix[1]
        return [a*x + b*y, c*x + d*y]

    @staticmethod
    def rotation(theta):
        return LinearTransform2D([
            [math.cos(theta), -math.sin(theta)],
            [math.sin(theta),  math.cos(theta)]
        ])

    @staticmethod
    def scaling(sx, sy):
        return LinearTransform2D([
            [sx, 0],
            [0, sy]
        ])

point = [1, 0]
R = LinearTransform2D.rotation(math.pi / 2)
S = LinearTransform2D.scaling(3, 2)

print("90도 회전:", R.apply(point))
print("스케일링:", S.apply(point))

7. 확률분포 클래스: 이산확률변수의 기댓값과 분산 🎲

제시 코드

class DiscreteDistribution:
    def __init__(self, outcomes):
        # outcomes: {값: 확률}
        self.outcomes = outcomes

    def expected_value(self):
        return sum(x * p for x, p in self.outcomes.items())

    def variance(self):
        mean = self.expected_value()
        return sum((x - mean)**2 * p for x, p in self.outcomes.items())

    def is_valid(self):
        return abs(sum(self.outcomes.values()) - 1) < 1e-9

dice = DiscreteDistribution({
    1: 1/6, 2: 1/6, 3: 1/6,
    4: 1/6, 5: 1/6, 6: 1/6
})

print("확률분포가 올바른가?", dice.is_valid())
print("기댓값:", dice.expected_value())
print("분산:", dice.variance())

8. 그래프 이론 클래스: BFS 최단거리 탐색 🕸️

제시 코드

from collections import deque

class Graph:
    def __init__(self):
        self.adj = {}

    def add_edge(self, a, b):
        self.adj.setdefault(a, []).append(b)
        self.adj.setdefault(b, []).append(a)

    def bfs_distance(self, start):
        distance = {start: 0}
        queue = deque([start])

        while queue:
            now = queue.popleft()
            for nxt in self.adj.get(now, []):
                if nxt not in distance:
                    distance[nxt] = distance[now] + 1
                    queue.append(nxt)

        return distance

g = Graph()
g.add_edge("A", "B")
g.add_edge("A", "C")
g.add_edge("B", "D")
g.add_edge("C", "E")

print(g.bfs_distance("A"))

class NewtonSolver:
    def __init__(self, function, derivative):
        self.f = function
        self.df = derivative

    def solve(self, x0, iterations=10):
        x = x0
        history = []

        for _ in range(iterations):
            x = x - self.f(x) / self.df(x)
            history.append(x)

        return x, history

def f(x):
    return x**2 - 2

def df(x):
    return 2*x

solver = NewtonSolver(f, df)
root, history = solver.solve(x0=1)

print("근사근:", root)
print("반복 과정:", history)

class EulerSolver:
    def __init__(self, derivative_function, t0, y0, h):
        self.f = derivative_function
        self.t = t0
        self.y = y0
        self.h = h

    def step(self):
        self.y = self.y + self.h * self.f(self.t, self.y)
        self.t = self.t + self.h
        return self.t, self.y

    def solve(self, steps):
        result = []
        for _ in range(steps):
            result.append(self.step())
        return result

def dydt(t, y):
    return y

solver = EulerSolver(dydt, t0=0, y0=1, h=0.1)

for t, y in solver.solve(10):
    print(round(t, 2), round(y, 4))

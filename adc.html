1. 원자와 분자량 계산 클래스 ⚛️

제시 코드

class Molecule:
    atomic_mass = {
        "H": 1.008, "C": 12.011, "N": 14.007,
        "O": 15.999, "Na": 22.990, "Cl": 35.45,
        "S": 32.06, "K": 39.098
    }

    def __init__(self, formula_dict):
        self.formula = formula_dict

    def molar_mass(self):
        total = 0
        for element, count in self.formula.items():
            total += Molecule.atomic_mass[element] * count
        return total

    def formula_string(self):
        return "".join(
            element + (str(count) if count > 1 else "")
            for element, count in self.formula.items()
        )

water = Molecule({"H": 2, "O": 1})
glucose = Molecule({"C": 6, "H": 12, "O": 6})

print(water.formula_string(), water.molar_mass())
print(glucose.formula_string(), glucose.molar_mass())

2. 몰수·질량·입자 수 변환 🧮

제시 코드

class ChemicalAmount:
    AVOGADRO = 6.022e23

    def __init__(self, molar_mass):
        self.molar_mass = molar_mass

    def mass_to_moles(self, mass):
        return mass / self.molar_mass

    def moles_to_mass(self, moles):
        return moles * self.molar_mass

    def moles_to_particles(self, moles):
        return moles * ChemicalAmount.AVOGADRO

co2 = ChemicalAmount(molar_mass=44.01)

moles = co2.mass_to_moles(88.02)
particles = co2.moles_to_particles(moles)

print("몰수:", moles)
print("입자 수:", particles)

3. 제한 반응물 계산 모델 ⚖️

제시 코드

class Reaction:
    def __init__(self, reactants):
        # reactants: {"물질명": (몰수, 계수)}
        self.reactants = reactants

    def limiting_reactant(self):
        ratios = {}
        for name, (moles, coefficient) in self.reactants.items():
            ratios[name] = moles / coefficient
        return min(ratios, key=ratios.get)

    def reaction_extent(self):
        limiter = self.limiting_reactant()
        moles, coefficient = self.reactants[limiter]
        return moles / coefficient

reaction = Reaction({
    "H2": (5.0, 2),
    "O2": (1.5, 1)
})

print("제한 반응물:", reaction.limiting_reactant())
print("최대 반응 진행량:", reaction.reaction_extent())

4. 이상기체 상태 방정식 모델 🎈

제시 코드

class IdealGas:
    R = 0.082057  # L·atm/(mol·K)

    def __init__(self, moles, temperature):
        self.n = moles
        self.T = temperature

    def pressure(self, volume_liter):
        return self.n * IdealGas.R * self.T / volume_liter

    def volume(self, pressure_atm):
        return self.n * IdealGas.R * self.T / pressure_atm

gas = IdealGas(moles=1.0, temperature=298)

print("22.4 L에서 압력:", gas.pressure(22.4), "atm")
print("1 atm에서 부피:", gas.volume(1.0), "L")

5. 산·염기 pH 계산 클래스 🧪

제시 코드

import math

class AcidSolution:
    def __init__(self, h_concentration):
        self.h = h_concentration

    def pH(self):
        return -math.log10(self.h)

    def pOH(self):
        return 14 - self.pH()

    def classification(self):
        if self.pH() < 7:
            return "산성"
        elif self.pH() == 7:
            return "중성"
        else:
            return "염기성"

solution = AcidSolution(1e-3)

print("pH:", solution.pH())
print("pOH:", solution.pOH())
print("분류:", solution.classification())

6. 완충 용액 Henderson–Hasselbalch 모델 🧴

제시 코드

import math

class BufferSolution:
    def __init__(self, pKa, acid_conc, base_conc):
        self.pKa = pKa
        self.acid = acid_conc
        self.base = base_conc

    def pH(self):
        return self.pKa + math.log10(self.base / self.acid)

    def add_acid(self, amount):
        self.acid += amount
        self.base -= amount

    def add_base(self, amount):
        self.base += amount
        self.acid -= amount

buffer = BufferSolution(pKa=4.76, acid_conc=0.10, base_conc=0.10)

print("초기 pH:", buffer.pH())
buffer.add_acid(0.02)
print("산 추가 후 pH:", buffer.pH())

7. 화학 평형과 반응지수 Q 모델 ⚖️

제시 코드

class EquilibriumReaction:
    def __init__(self, K, reactants, products):
        # {"A": (농도, 계수)}
        self.K = K
        self.reactants = reactants
        self.products = products

    def reaction_quotient(self):
        numerator = 1
        denominator = 1

        for conc, coeff in self.products.values():
            numerator *= conc ** coeff

        for conc, coeff in self.reactants.values():
            denominator *= conc ** coeff

        return numerator / denominator

    def direction(self):
        Q = self.reaction_quotient()
        if Q < self.K:
            return "정반응 방향"
        elif Q > self.K:
            return "역반응 방향"
        else:
            return "평형 상태"

eq = EquilibriumReaction(
    K=10,
    reactants={"A": (0.5, 1), "B": (0.5, 1)},
    products={"C": (1.0, 1)}
)

print("Q:", eq.reaction_quotient())
print("반응 방향:", eq.direction())

8. 반응 속도식과 속도 상수 모델 ⏱️

제시 코드

class RateLaw:
    def __init__(self, k, orders):
        # orders: {"A": 1, "B": 2}
        self.k = k
        self.orders = orders

    def rate(self, concentrations):
        result = self.k
        for species, order in self.orders.items():
            result *= concentrations[species] ** order
        return result

    def overall_order(self):
        return sum(self.orders.values())

law = RateLaw(k=0.25, orders={"A": 1, "B": 2})

concentrations = {"A": 0.4, "B": 0.2}

print("반응 속도:", law.rate(concentrations))
print("전체 반응 차수:", law.overall_order())

9. 아레니우스 식과 온도 의존성 🔥

제시 코드

import math

class ArrheniusModel:
    R = 8.314

    def __init__(self, frequency_factor, activation_energy):
        self.A = frequency_factor
        self.Ea = activation_energy

    def rate_constant(self, temperature):
        return self.A * math.exp(-self.Ea / (ArrheniusModel.R * temperature))

    def ratio(self, T1, T2):
        return self.rate_constant(T2) / self.rate_constant(T1)

model = ArrheniusModel(frequency_factor=1e12, activation_energy=75000)

print("300K 속도상수:", model.rate_constant(300))
print("350K 속도상수:", model.rate_constant(350))
print("속도상수 증가 비:", model.ratio(300, 350))

10. 전기화학 Nernst 식 모델 🔋

제시 코드

import math

class ElectrochemicalCell:
    def __init__(self, standard_potential, electrons):
        self.E0 = standard_potential
        self.n = electrons

    def potential(self, Q):
        return self.E0 - (0.0592 / self.n) * math.log10(Q)

    def spontaneity(self, Q):
        E = self.potential(Q)
        return "자발적" if E > 0 else "비자발적"

cell = ElectrochemicalCell(standard_potential=1.10, electrons=2)

for Q in [0.1, 1, 10, 100]:
    print("Q =", Q, "전지 전위:", round(cell.potential(Q), 3), "V", cell.spontaneity(Q))

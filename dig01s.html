class PrivacyChecklist:
    def __init__(self, practices):
        self.practices = practices
    def score(self):
        good_rules = ["비밀번호 주기적 변경", "개인 정보 친구에게 알리지 않기", "P2P 폴더에 개인 정보 저장하지 않기", "출처 불명 자료 다운로드하지 않기", "자동 로그인 해제"]
        return sum(1 for rule in good_rules if rule in self.practices)
    def level(self):
        s = self.score()
        if s >= 4: return "우수"
        if s >= 2: return "보통"
        return "개선 필요"
practices = ["비밀번호 주기적 변경", "출처 불명 자료 다운로드하지 않기", "자동 로그인 해제"]
check = PrivacyChecklist(practices)
print("실천 점수:", check.score())
print("보호 수준:", check.level())

class PasswordStrengthChecker:
    def __init__(self, password):
        self.password = password
    def score(self):
        score = 0
        if len(self.password) >= 8: score += 1
        if any(c.isdigit() for c in self.password): score += 1
        if any(c.isupper() for c in self.password): score += 1
        if any(not c.isalnum() for c in self.password): score += 1
        return score
    def result(self):
        if self.score() >= 3: return "강한 비밀번호"
        if self.score() == 2: return "보통 비밀번호"
        return "약한 비밀번호"
checker = PasswordStrengthChecker("Data!2026")
print("점수:", checker.score())
print("결과:", checker.result())

class AppPermissionChecker:
    def __init__(self, app_name, permissions):
        self.app_name = app_name
        self.permissions = permissions
    def risky_permissions(self):
        risky = ["위치", "연락처", "카메라", "마이크", "저장소"]

    def advice(self):
        risky_count = len(self.risky_permissions())
        if risky_count >= 3: return "권한 설정을 자세히 점검하세요."
        if risky_count >= 1: return "필요한 권한인지 확인하세요."
        return "위험 권한이 적습니다."
app = AppPermissionChecker("메모 앱", ["저장소", "위치", "마이크"])

class PhishingDetector:
    def __init__(self, text):
        self.text = text
    def risk_score(self):
        keywords = ["비밀번호", "계좌", "즉시", "링크", "개인 정보", "인증"]
        return sum(1 for keyword in keywords if keyword in self.text)
    def classify(self):
        if self.risk_score() >= 3: return "피싱 위험 높음"
        if self.risk_score() >= 1: return "주의 필요"
        return "위험 낮음"
msg = PhishingDetector("즉시 링크에 접속해 계좌 인증과 비밀번호를 입력하세요.")
print("위험 점수:", msg.risk_score())

class CCTVAccessLog:
    def __init__(self):
        self.logs = []
    def add_access(self, user, purpose):
        self.logs.append({"사용자": user, "목적": purpose})
    def allowed_access(self):
        allowed = ["범죄 예방", "안전 관리", "교통 관리"]
        return [log for log in self.logs if log["목적"] in allowed]
    def suspicious_access(self):
        allowed = ["범죄 예방", "안전 관리", "교통 관리"]
        return [log for log in self.logs if log["목적"] not in allowed]
log = CCTVAccessLog()
log.add_access("관리자A", "안전 관리")
log.add_access("사용자B", "개인 호기심")
print("허용 접근:", log.allowed_access())
print("의심 접근:", log.suspicious_access())

class SecurityPracticeDashboard:
    def __init__(self, user_name):
        self.user_name = user_name
        self.items = {}
    def set_item(self, item, done):
        self.items[item] = done
    def completion_rate(self):
        if not self.items: return 0
        done_count = sum(1 for done in self.items.values() if done)
        return done_count / len(self.items) * 100
    def report(self):
        rate = self.completion_rate()
        if rate >= 80: level = "우수"
        elif rate >= 50: level = "보통"
        else: level = "개선 필요"
        return f"{self.user_name}의 정보 보안 실천율은 {rate:.1f}%이며 수준은 {level}입니다."
dashboard = SecurityPracticeDashboard("김학생")
dashboard.set_item("비밀번호 주기적 변경", True)
dashboard.set_item("백신 업데이트", True)
dashboard.set_item("불필요한 개인 정보 삭제", False)
dashboard.set_item("출처 불명 링크 클릭 금지", True)
print(dashboard.report())

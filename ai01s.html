class RuleBasedClassifier:
    def __init__(self):
        self.rules = []

    def add_rule(self, condition, label):
        self.rules.append((condition, label))

    def predict(self, data):
        for condition, label in self.rules:
            if condition(data):
                return label
        return "판단 불가"

classifier = RuleBasedClassifier()
classifier.add_rule(lambda x: x["온도"] >= 30, "더움")
classifier.add_rule(lambda x: x["온도"] <= 10, "추움")
print(classifier.predict({"온도": 35}))

6. 단순 회귀 모델: 공부 시간으로 점수 예측

제시 코드

class SimpleRegression:
    def __init__(self):
        self.a = 0
        self.b = 0

    def fit(self, X, y):
        x_mean = sum(X) / len(X)
        y_mean = sum(y) / len(y)
        numerator = sum((x - x_mean) * (target - y_mean) for x, target in zip(X, y))
        denominator = sum((x - x_mean) ** 2 for x in X)
        self.a = numerator / denominator
        self.b = y_mean - self.a * x_mean

    def predict(self, x):
        return self.a * x + self.b

hours = [1, 2, 3, 4, 5]
scores = [50, 60, 65, 75, 85]
model = SimpleRegression()
model.fit(hours, scores)
print("6시간 공부 예측 점수:", model.predict(6))

7. KNN 분류 모델: 가까운 이웃으로 종류 판단

제시 코드

class KNNClassifier:
    def __init__(self, k=3):
        self.k = k
        self.X_train = []
        self.y_train = []

    def fit(self, X, y):
        self.X_train = X
        self.y_train = y

    def distance(self, a, b):
        return sum((x - y) ** 2 for x, y in zip(a, b)) ** 0.5

    def predict_one(self, x):
        distances = [(self.distance(x, tx), label) for tx, label in zip(self.X_train, self.y_train)]
        distances.sort()
        nearest = distances[:self.k]
        votes = {}
        for _, label in nearest:
            votes[label] = votes.get(label, 0) + 1
        return max(votes, key=votes.get)

X = [[1, 1], [2, 1], [8, 8], [9, 8]]
y = ["고양이", "고양이", "강아지", "강아지"]
model = KNNClassifier(k=3)
model.fit(X, y)
print(model.predict_one([2, 2]))

class SimpleClusterer:
    def __init__(self, threshold):
        self.threshold = threshold

    def cluster(self, students):
        active, passive = [], []
        for student in students:
            score = student["발표횟수"] + student["자료조회"]
            if score >= self.threshold:
                active.append(student["이름"])
            else:
                passive.append(student["이름"])
        return {"적극 참여 그룹": active, "보통 참여 그룹": passive}

students = [
    {"이름": "김", "발표횟수": 10, "자료조회": 30},
    {"이름": "이", "발표횟수": 2, "자료조회": 8},
    {"이름": "박", "발표횟수": 8, "자료조회": 25}
]
print(SimpleClusterer(threshold=30).cluster(students))

class AccuracyEvaluator:
    def __init__(self, y_true, y_pred):
        self.y_true = y_true
        self.y_pred = y_pred

    def accuracy(self):
        correct = 0
        for true, pred in zip(self.y_true, self.y_pred):
            if true == pred:
                correct += 1
        return correct / len(self.y_true)

true_labels = ["A", "B", "A", "C", "B"]
pred_labels = ["A", "B", "C", "C", "B"]

class MinMaxScalerSimple:
    def __init__(self, data):
        self.data = data

    def normalize(self):
        min_value = min(self.data)
        max_value = max(self.data)
        return [(x - min_value) / (max_value - min_value) for x in self.data]

class CO2EmissionPredictor:
    def __init__(self, engine_weight=40, fuel_weight=20, bias=50):
        self.engine_weight = engine_weight
        self.fuel_weight = fuel_weight
        self.bias = bias

    def predict(self, engine_size, fuel_consumption):

class SNSClusterRecommender:
    def __init__(self, users):
        self.users = users

    def cluster_users(self):
        clusters = {"운동 관심 그룹": [], "음악 관심 그룹": [], "학습 관심 그룹": []}
        for user in self.users:
            if user["운동"] >= user["음악"] and user["운동"] >= user["학습"]:
                clusters["운동 관심 그룹"].append(user["이름"])
            elif user["음악"] >= user["운동"] and user["음악"] >= user["학습"]:
                clusters["음악 관심 그룹"].append(user["이름"])
            else:
                clusters["학습 관심 그룹"].append(user["이름"])
        return clusters

users = [
    {"이름": "김", "운동": 9, "음악": 3, "학습": 5},
    {"이름": "이", "운동": 2, "음악": 8, "학습": 4},
    {"이름": "박", "운동": 4, "음악": 5, "학습": 9}
]
print(SNSClusterRecommender(users).cluster_users())

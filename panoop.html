class MissingValueHandler:
    def __init__(self, df):
        self.df = df.copy()

    def fill_with_mean(self, column):
        mean_value = self.df[column].mean()
        self.df[column] = self.df[column].fillna(mean_value)
        return self.df

    def drop_missing(self):
        return self.df.dropna()

df = pd.DataFrame({

import pandas as pd

class DataMerger:
    def __init__(self, left, right):
        self.left = left
        self.right = right

    def inner_merge(self, key):
        return pd.merge(self.left, self.right, on=key, how="inner")

    def left_merge(self, key):
        return pd.merge(self.left, self.right, on=key, how="left")

students = pd.DataFrame({
    "학생ID": [1, 2, 3],
    "이름": ["김", "이", "박"]
})

scores = pd.DataFrame({
    "학생ID": [1, 2, 4],
    "점수": [90, 85, 70]

merger = DataMerger(students, scores)

import pandas as pd

class TimeSeriesAggregator:
    def __init__(self, df, date_col):
        self.df = df.copy()
        self.date_col = date_col
        self.df[date_col] = pd.to_datetime(self.df[date_col])

    def monthly_sum(self, value_col):
        temp = self.df.set_index(self.date_col)
        return temp[value_col].resample("M").sum()

df = pd.DataFrame({
    "날짜": ["2026-01-01", "2026-01-15", "2026-02-01", "2026-02-20"],

import pandas as pd

class SalesPipeline:
    def __init__(self, data):
        self.df = pd.DataFrame(data)

    def clean(self):
        self.df["지역"] = self.df["지역"].str.strip()
        self.df["매출"] = self.df["매출"].fillna(self.df["매출"].mean())
        return self

    def add_grade(self):
        self.df["등급"] = self.df["매출"].apply(lambda x: "높음" if x >= 100 else "낮음")
        return self

    def summarize(self):
        return self.df.groupby("지역")["매출"].agg(["count", "mean", "sum"])

    def result(self):
        return self.df

data = {
    "지역": ["서울 ", "서울", "부산", "부산", "대구"],
    "매출": [120, None, 80, 150, 90]
}

pipeline = SalesPipeline(data)
summary = pipeline.clean().add_grade().summarize()

print(pipeline.result())
print(summary)

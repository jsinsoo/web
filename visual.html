월별 매출 꺾은선그래프

출력·활용 설명: 월을 x축, 매출을 y축에 배치한 기본 시계열 그래프이다. marker는 실제 관측 지점을 보여 주고, grid는 값의 변화를 읽기 쉽게 한다.

from pathlib import Path
import matplotlib.pyplot as plt
from matplotlib import font_manager

# Windows 한글 글꼴 자동 설정
font_candidates = [
    Path("C:/Windows/Fonts/malgun.ttf"),
    Path("C:/Windows/Fonts/malgunbd.ttf"),
]
font_path = next((p for p in font_candidates if p.exists()), None)

if font_path:
    font_name = font_manager.FontProperties(fname=str(font_path)).get_name()
    plt.rcParams["font.family"] = font_name
else:
    print("맑은 고딕 글꼴을 찾지 못했습니다. Windows 글꼴 설치 상태를 확인하세요.")

plt.rcParams["axes.unicode_minus"] = False

months = ["1월", "2월", "3월", "4월", "5월", "6월"]
sales = [120, 135, 128, 150, 170, 190]

plt.figure(figsize=(8, 4.8))
plt.plot(months, sales, marker="o", linewidth=2)
plt.title("월별 매출 변화")
plt.xlabel("월")
plt.ylabel("매출(만원)")
plt.grid(True, linestyle="--", alpha=0.4)
plt.tight_layout()
plt.show()

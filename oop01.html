OOP 파이썬 코드

class HumanSentence:
    def __init__(self, text):
        self.text = text
        self.words = text.replace(",", "").replace(".", "").split()

    def edges(self):
        return self.words[0], self.words[-1]

    def front(self, n):
        return self.words[:n]

sentence = HumanSentence("To ask what a human being is, we must study the body, the mind, and society together.")
first, last = sentence.edges()

print("첫 단어:", first)
print("마지막 단어:", last)
print("앞 6개:", sentence.front(6))

상세 풀이

1단계: 문장을 split()으로 단어 리스트로 바꿉니다.

2단계: words[start:end:step] 형식으로 앞부분, 뒷부분, 역순, 일정 구간을 추출합니다.

3단계: 클래스와 메서드를 사용하므로 같은 구조를 다른 문장에도 재사용할 수 있습니다.

class BilingualSentence:
    def __init__(self, english, korean):
        self.english_words = english.replace(",", "").replace(".", "").split()

    def pair_words(self, n):

OOP 파이썬 코드

class PopPractice:
    def __init__(self, text):
        self.words = text.replace(".", "").split()

    def pop_until_three(self):
        removed = []
        while len(self.words) > 3:
            removed.append(self.words.pop())
        return removed, self.words

obj = PopPractice("Xunzi argued that human desires can become selfish unless they are shaped by education and rules.")
removed, remain = obj.pop_until_three()

print("꺼낸 단어:", removed[:5])
print("남은 단어:", remain)

Freud explored the hidden forces of the unconscious that influence human thought and behavior.

핵심 학습: 이 문장을 단어 리스트로 바꾸고, OOP 클래스 안에서 리스트 슬라이싱과 파이썬 문법을 함께 연습합니다.

OOP 파이썬 코드

class KeywordFinder:
    def __init__(self, text):
        self.words = text.replace(".", "").split()
        self.lower_words = [w.lower() for w in self.words]

    def around(self, keyword):
        if keyword in self.lower_words:
            idx = self.lower_words.index(keyword)
            return self.words[max(0, idx-2):idx+3]
        return self.words[-5:]

obj = KeywordFinder("Freud explored the hidden forces of the unconscious that influence human thought and behavior.")

print("human 주변:", obj.around("human"))
print("unconscious 주변:", obj.around("unconscious"))

OOP 파이썬 코드

class WhileSlicer:
    def __init__(self, text):
        self.words = text.replace(".", "").split()

    def split_by_while(self, size):
        start = 0
        result = []
        while start < len(self.words):
            result.append(self.words[start:start+size])
            start += size
        return result

obj = WhileSlicer("He showed that the self is not always transparent to itself.")

print(obj.split_by_while(2))

OOP 파이썬 코드

class KwargsSlicer:
    def slice_words(self, **kwargs):
        text = kwargs["text"]
        start = kwargs.get("start", 0)
        end = kwargs.get("end", 5)
        step = kwargs.get("step", 1)
        words = text.replace(",", "").replace(".", "").split()

class TotalTransformer:
    def transform(self, *sentences, **kwargs):
        start = kwargs.get("start", 0)
        end = kwargs.get("end", 5)
        reverse = kwargs.get("reverse", False)
        results = []

        for sentence in sentences:
            words = sentence.replace(",", "").replace(".", "").split()
            piece = words[start:end]
            if reverse:
                piece = piece[::-1]
            results.append("-".join(piece))
        return results

obj = TotalTransformer()

    "Human enhancement technologies raise ethical questions about health, fairness, and human limits.",
    "Ethics asks how technology should serve humans.",
    start=0,
    end=6,
    reverse=True
))

        sorted_words = sorted(self.words, key=lambda w: (len(w), w.lower()))
        long_words = list(filter(lambda w: len(w) >= 5, self.words))
        pairs = list(zip(self.words[:5], list(map(len, self.words[:5]))))

        result["단어와 길이"] = pairs

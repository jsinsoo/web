상세 설명: pandas DataFrame의 행을 반복하며 마커를 표시한다. CSV 파일에서 읽어 온 장소 데이터를 지도화할 때 자주 쓰는 구조이다.

활용 포인트: iterrows()는 각 행의 값을 꺼내 지도에 추가할 수 있게 해 준다.

import folium
import pandas as pd

df = pd.DataFrame({
    '이름':['서울시청','경복궁','남산타워'],
    '위도':[37.5665,37.5796,37.5512],
    '경도':[126.9780,126.9770,126.9882]
})
m = folium.Map(location=[37.5665,126.9780], zoom_start=12)
for _, row in df.iterrows():
    folium.Marker([row['위도'], row['경도']], popup=row['이름']).add_to(m)
m.save('22_dataframe_marker.html')

import pandas as pd

    '장소':['A','B','C'], '위도':[37.56,37.57,37.58], '경도':[126.97,126.98,126.99], '혼잡도':[30,80,55]
})
m = folium.Map(location=[37.57,126.98], zoom_start=13)
for _, r in df.iterrows():
    color = 'red' if r['혼잡도'] >= 70 else 'green'
    folium.Marker([r['위도'],r['경도']], popup=f"{r['장소']} 혼잡도:{r['혼잡도']}", icon=folium.Icon(color=color)).add_to(m)

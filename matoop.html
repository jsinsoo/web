class MatrixMultiplier:
    def __init__(self, A, B):
        self.A = A
        self.B = B

    def multiply(self):
        result = []
        for i in range(len(self.A)):
            row = []
            for j in range(len(self.B[0])):
                s = 0
                for k in range(len(self.B)):
                    s += self.A[i][k] * self.B[k][j]
                row.append(s)
            result.append(row)
        return result

제시 코드

class Inverse2x2:
    def __init__(self, A):
        self.A = A

    def inverse(self):
        a, b = self.A[0]
        c, d = self.A[1]
        det = a * d - b * c
        if det == 0:
            return None
        return [[d / det, -b / det],
                [-c / det, a / det]]

A = [[2, 3], [1, 4]]

13. 역행렬로 연립방정식 풀기 LinearSystem2x2

제시 코드

class LinearSystem2x2:
    def __init__(self, A, b):
        self.A = A
        self.b = b

    def inverse(self):
        a, b = self.A[0]
        c, d = self.A[1]
        det = a * d - b * c
        return [[d / det, -b / det], [-c / det, a / det]]

    def solve(self):
        inv = self.inverse()
        x = inv[0][0] * self.b[0] + inv[0][1] * self.b[1]
        y = inv[1][0] * self.b[0] + inv[1][1] * self.b[1]
        return x, y

A = [[2, 1], [1, 3]]
b = [5, 6]
system = LinearSystem2x2(A, b)
print(system.solve())

14. 행렬 거듭제곱 MatrixPower

제시 코드

class MatrixPower:
    def __init__(self, A):
        self.A = A

    def multiply(self, X, Y):
        return [[sum(X[i][k] * Y[k][j] for k in range(len(Y)))
                 for j in range(len(Y[0]))]
                 for i in range(len(X))]

    def power(self, n):
        result = self.A
        for _ in range(n - 1):
            result = self.multiply(result, self.A)
        return result

A = [[1, 1], [1, 0]]
mp = MatrixPower(A)

17. 가우스-조르당 역행렬 GaussJordanInverse2x2

제시 코드

class GaussJordanInverse2x2:
    def __init__(self, A):
        self.A = [row[:] for row in A]

    def inverse(self):
        n = 2
        I = [[1, 0], [0, 1]]
        aug = [self.A[i] + I[i] for i in range(n)]

        for i in range(n):
            pivot = aug[i][i]
            for j in range(2 * n):
                aug[i][j] /= pivot

            for r in range(n):
                if r != i:
                    factor = aug[r][i]
                    for c in range(2 * n):
                        aug[r][c] -= factor * aug[i][c]

        return [row[n:] for row in aug]

A = [[2, 1], [1, 1]]
g = GaussJordanInverse2x2(A)
print(g.inverse())

18. 간단한 랭크 계산 MatrixRankSimple

제시 코드

class MatrixRankSimple:
    def __init__(self, A):
        self.A = [list(map(float, row)) for row in A]

    def rank(self):
        A = self.A
        rows, cols = len(A), len(A[0])
        r = 0
        for c in range(cols):
            pivot = None
            for i in range(r, rows):
                if A[i][c] != 0:
                    pivot = i
                    break
            if pivot is not None:
                A[r], A[pivot] = A[pivot], A[r]
                div = A[r][c]
                A[r] = [x / div for x in A[r]]
                for i in range(rows):
                    if i != r:
                        factor = A[i][c]
                        A[i] = [A[i][j] - factor * A[r][j] for j in range(cols)]
                r += 1
        return r

A = [[1, 2], [2, 4]]
ranker = MatrixRankSimple(A)
print(ranker.rank())

20. 행렬 분석 종합 클래스 MatrixPipeline

제시 코드

class MatrixPipeline:
    def __init__(self, A):
        self.A = A

    def shape(self):
        return len(self.A), len(self.A[0])

    def det2(self):
        a, b = self.A[0]
        c, d = self.A[1]
        return a * d - b * c

    def inverse2(self):
        det = self.det2()
        if det == 0:
            return "역행렬 없음"
        a, b = self.A[0]
        c, d = self.A[1]
        return [[d / det, -b / det], [-c / det, a / det]]

    def report(self):
        return {"크기": self.shape(), "행렬식": self.det2(), "역행렬": self.inverse2()}

A = [[3, 1], [2, 1]]
pipe = MatrixPipeline(A)
print(pipe.report())

1. 대기 기압의 고도 변화 모델 ☁️

제시 코드

import math

class Atmosphere:
    def __init__(self, sea_level_pressure=101325, scale_height=8400):
        self.p0 = sea_level_pressure
        self.H = scale_height

    def pressure_at_height(self, height):
        return self.p0 * math.exp(-height / self.H)

    def report(self, height):
        p = self.pressure_at_height(height)
        return f"고도 {height} m의 기압: {p:.2f} Pa"

atm = Atmosphere()

for h in [0, 1000, 3000, 8000]:
    print(atm.report(h))

제시 코드

class AtmosphericLayer:
    DRY_ADIABATIC_RATE = 9.8  # °C/km

    def __init__(self, lower_temp, upper_temp, height_diff_km):
        self.lower_temp = lower_temp
        self.upper_temp = upper_temp
        self.height_diff_km = height_diff_km

    def environmental_lapse_rate(self):
        return (self.lower_temp - self.upper_temp) / self.height_diff_km

    def stability(self):
        rate = self.environmental_lapse_rate()
        if rate > AtmosphericLayer.DRY_ADIABATIC_RATE:
            return "불안정"
        elif rate == AtmosphericLayer.DRY_ADIABATIC_RATE:
            return "중립"
        else:
            return "안정"

layer = AtmosphericLayer(lower_temp=25, upper_temp=5, height_diff_km=3)

print("환경 감률:", round(layer.environmental_lapse_rate(), 2), "°C/km")
print("대기 안정도:", layer.stability())

class Seawater:
    def __init__(self, temperature, salinity):
        self.temperature = temperature
        self.salinity = salinity

    def density(self):
        base = 1000
        salinity_effect = 0.8 * self.salinity
        temperature_effect = -0.2 * self.temperature
        return base + salinity_effect + temperature_effect

    def is_dense_water(self):
        return self.density() > 1025

water = Seawater(temperature=2, salinity=35)

print("해수 밀도:", water.density(), "kg/m^3")
print("고밀도 수괴인가?", water.is_dense_water())

class OceanParticle:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def move(self, current_u, current_v, dt):
        self.x += current_u * dt
        self.y += current_v * dt

    def position(self):
        return round(self.x, 2), round(self.y, 2)

particle = OceanParticle(x=0, y=0)

currents = [(0.5, 0.1), (0.4, 0.2), (0.3, 0.3), (0.2, 0.4)]

for u, v in currents:
    particle.move(current_u=u, current_v=v, dt=1)
    print("입자 위치:", particle.position())

    def __init__(self, magnitude, depth_km, location):
        self.magnitude = magnitude
        self.depth_km = depth_km

        return 10 ** (1.5 * self.magnitude + 4.8)

        elif self.magnitude >= 5:
            return "주의"
        else:
            return "낮음"

class TectonicPlate:
    def __init__(self, name, speed_cm_per_year, direction):
        self.name = name
        self.speed = speed_cm_per_year
        self.direction = direction

    def displacement(self, years):
        return self.speed * years / 100  # cm를 m로 변환

    def report(self, years):
        d = self.displacement(years)
        return f"{self.name}판은 {years}년 동안 {self.direction} 방향으로 {d:.2f} m 이동"

plate = TectonicPlate("태평양", speed_cm_per_year=8, direction="북서쪽")

print(plate.report(1000))
print(plate.report(1_000_000))

class Volcano:
    def __init__(self, name, magma_pressure, gas_level, seismic_activity):
        self.name = name
        self.magma_pressure = magma_pressure
        self.gas_level = gas_level
        self.seismic_activity = seismic_activity

    def risk_score(self):
        return self.magma_pressure * 0.4 + self.gas_level * 0.3 + self.seismic_activity * 0.3

    def alert_level(self):
        score = self.risk_score()
        if score >= 80:
            return "경보"
        elif score >= 50:
            return "주의"
        else:
            return "평상"

volcano = Volcano("한라산 가상 모델", magma_pressure=65, gas_level=70, seismic_activity=40)

print("위험 점수:", volcano.risk_score())
print("경보 단계:", volcano.alert_level())

8. 탄소 순환 상자 모델 🌱🏭

제시 코드

class CarbonBoxModel:
    def __init__(self, atmosphere, ocean, land):
        self.atmosphere = atmosphere
        self.ocean = ocean
        self.land = land

    def step(self, emission, ocean_absorption, land_absorption):
        self.atmosphere += emission
        self.atmosphere -= ocean_absorption + land_absorption
        self.ocean += ocean_absorption
        self.land += land_absorption

    def state(self):
        return {
            "atmosphere": round(self.atmosphere, 2),
            "ocean": round(self.ocean, 2),
            "land": round(self.land, 2)
        }

model = CarbonBoxModel(atmosphere=850, ocean=38000, land=2500)

for year in range(5):
    model.step(emission=10, ocean_absorption=3, land_absorption=2)
    print(year + 1, "년 후:", model.state())

9. 별의 광도와 겉보기 밝기 모델 ✨

import math

class Star:
    def __init__(self, name, luminosity, distance):
        self.name = name
        self.luminosity = luminosity
        self.distance = distance

    def flux(self):
        return self.luminosity / (4 * math.pi * self.distance ** 2)

    def compare_brightness(self, other):
        return self.flux() / other.flux()

sun_like = Star("가까운 별", luminosity=3.8e26, distance=1.5e11)
far_star = Star("먼 별", luminosity=3.8e28, distance=1.5e13)

print(sun_like.name, "복사 플럭스:", sun_like.flux())
print(far_star.name, "복사 플럭스:", far_star.flux())
print("밝기 비:", sun_like.compare_brightness(far_star))

10. 행성의 평형 온도 모델 🪐🌡️

제시 코드

import math

class PlanetClimate:
    SIGMA = 5.67e-8

    def __init__(self, star_luminosity, distance, albedo):
        self.star_luminosity = star_luminosity
        self.distance = distance
        self.albedo = albedo

    def equilibrium_temperature(self):
        numerator = self.star_luminosity * (1 - self.albedo)
        denominator = 16 * math.pi * PlanetClimate.SIGMA * self.distance ** 2
        return (numerator / denominator) ** 0.25

    def celsius(self):
        return self.equilibrium_temperature() - 273.15

earth_model = PlanetClimate(star_luminosity=3.828e26, distance=1.496e11, albedo=0.30)

print("평형 온도:", earth_model.equilibrium_temperature(), "K")
print("섭씨:", earth_model.celsius(), "°C")

예제 1: Visual Analysis - Graph 25

핵심 그래프: KDE

① 원본 코드

import seaborn as sns
import matplotlib.pyplot as plt

# 한글 폰트 설정 예시 (Windows 기준)
plt.rcParams['font.family'] = 'Malgun Gothic'
plt.rcParams['axes.unicode_minus'] = False

# 1. 데이터 로드 및 스타일 설정
tips = sns.load_dataset("tips")
sns.set_style("ticks")

# 2. 2D KDE 플롯 그리기
plt.figure(figsize=(10, 7))
sns.kdeplot(data=tips, x="total_bill", y="tip",
            thresh=0.05,       # 하위 5% 밀도 영역은 제외 (선택 사항)
            levels=10,         # 등고선 층의 개수
            color="tab:blue")  # 선 색상

# 3. 제목 및 라벨 설정
plt.title("Visual Analysis - Graph 25")
plt.xlabel("total_bill")
plt.ylabel("tip")

plt.show()

import matplotlib.pyplot as plt

sns.set_style("ticks")

# 2행 3열의 서브플롯 구성 (마지막 칸은 비워둠)
fig, axes = plt.subplots(2, 3, figsize=(20, 12))

# Graph 3: 바이올린 플롯 (밀도 추정)
sns.violinplot(data=tips, x="time", y="tip", ax=axes[0, 2], color="tab:blue")
axes[0, 2].set_title("Graph 3: Violin Plot")

# Graph 4: 그룹화된 산점도 (성별 구분)
sns.scatterplot(data=tips, x="total_bill", y="tip", hue="sex", ax=axes[1, 0])
axes[1, 0].set_title("Graph 4: Scatter with Hue")

# Graph 5: 막대 그래프 (평균 및 신뢰구간)
sns.barplot(data=tips, x="day", y="tip", ax=axes[1, 1], color="tab:blue")
axes[1, 1].set_title("Graph 5: Bar Plot")

# 마지막 빈 칸 제거 및 레이아웃 조정
fig.delaxes(axes[1, 2])
plt.tight_layout()
plt.show()

import seaborn as sns
import matplotlib.pyplot as plt

# 1. 데이터 로드 및 상관계수 계산
tips = sns.load_dataset("tips")
# 수치형 데이터만 선택하여 상관계수 행렬 생성
corr = tips[['total_bill', 'tip', 'size']].corr()

# 2. 히트맵 그리기
plt.figure(figsize=(10, 7))
sns.heatmap(corr,
            annot=True,       # 칸 안에 숫자 표시
            cmap='coolwarm',  # 빨간색(양)과 파란색(음)의 대비 색상
            fmt=".2g",        # 숫자 포맷 설정
            center=0.7)       # 색상 기준점 설정 (이미지와 유사하게 조정)

# 3. 제목 설정
plt.title("Visual Analysis - Graph 16")

plt.show()

1. 뉴턴 역학 입자 시뮬레이션 🏀

제시 코드

class Particle:
    def __init__(self, mass, position, velocity):
        self.mass = mass
        self.position = position
        self.velocity = velocity

    def acceleration(self, force):
        return force / self.mass

    def update(self, force, dt):
        a = self.acceleration(force)
        self.velocity += a * dt
        self.position += self.velocity * dt

p = Particle(mass=2.0, position=0.0, velocity=0.0)

for t in range(5):
    p.update(force=10.0, dt=1.0)
    print(t + 1, "초 후 위치:", p.position, "속도:", p.velocity)

class Projectile:
    def __init__(self, x, y, vx, vy, g=9.8):
        self.x = x
        self.y = y
        self.vx = vx
        self.vy = vy
        self.g = g

    def step(self, dt):
        self.vy -= self.g * dt
        self.x += self.vx * dt
        self.y += self.vy * dt

    def state(self):
        return (round(self.x, 2), round(self.y, 2), round(self.vy, 2))

ball = Projectile(x=0, y=0, vx=10, vy=30)

for i in range(6):
    ball.step(0.5)
    print("상태:", ball.state())

제시 코드

class SpringMass:
    def __init__(self, mass, k, x, v):
        self.mass = mass
        self.k = k
        self.x = x
        self.v = v

    def force(self):
        return -self.k * self.x

    def energy(self):
        kinetic = 0.5 * self.mass * self.v ** 2
        potential = 0.5 * self.k * self.x ** 2
        return kinetic + potential

    def step(self, dt):
        a = self.force() / self.mass
        self.v += a * dt
        self.x += self.v * dt

osc = SpringMass(mass=1.0, k=4.0, x=1.0, v=0.0)

for i in range(10):
    osc.step(0.1)
    print(i, "위치:", round(osc.x, 3), "에너지:", round(osc.energy(), 3))

class ChargedParticle:
    def __init__(self, charge, mass, x, v):
        self.charge = charge
        self.mass = mass
        self.x = x
        self.v = v

    def electric_force(self, electric_field):
        return self.charge * electric_field

    def step(self, electric_field, dt):
        force = self.electric_force(electric_field)
        a = force / self.mass
        self.v += a * dt
        self.x += self.v * dt

electron_like = ChargedParticle(charge=-1.0, mass=1.0, x=0.0, v=0.0)

for i in range(5):
    electron_like.step(electric_field=2.0, dt=0.1)

5. 쿨롱 힘을 받는 두 전하 시스템 🧲

제시 코드

class Charge:
    K = 8.99e9

    def __init__(self, q, x):
        self.q = q
        self.x = x

    def force_from(self, other):
        r = self.x - other.x
        direction = 1 if r > 0 else -1
        distance = abs(r)
        return Charge.K * self.q * other.q / distance**2 * direction

q1 = Charge(1e-6, 0.0)
q2 = Charge(-2e-6, 0.5)

print("q1이 받는 힘:", q1.force_from(q2), "N")
print("q2가 받는 힘:", q2.force_from(q1), "N")

class Planet:
    G = 6.674e-11

    def __init__(self, name, mass, orbit_radius):
        self.name = name
        self.mass = mass
        self.orbit_radius = orbit_radius

    def orbital_speed(self, central_mass):
        return (Planet.G * central_mass / self.orbit_radius) ** 0.5

    def potential_energy(self, central_mass):
        return -Planet.G * central_mass * self.mass / self.orbit_radius

earth = Planet("Earth", 5.97e24, 1.496e11)
sun_mass = 1.989e30

print("공전 속도:", earth.orbital_speed(sun_mass), "m/s")
print("중력 퍼텐셜 에너지:", earth.potential_energy(sun_mass), "J")

7. 상대론적 입자의 에너지 ⚛️

제시 코드

class RelativisticParticle:
    C = 299_792_458

    def __init__(self, rest_mass, speed):
        self.rest_mass = rest_mass
        self.speed = speed

    def gamma(self):
        beta2 = (self.speed / RelativisticParticle.C) ** 2
        return 1 / (1 - beta2) ** 0.5

    def total_energy(self):
        return self.gamma() * self.rest_mass * RelativisticParticle.C ** 2

    def kinetic_energy(self):
        rest_energy = self.rest_mass * RelativisticParticle.C ** 2
        return self.total_energy() - rest_energy

p = RelativisticParticle(rest_mass=9.11e-31, speed=2.0e8)

print("감마:", p.gamma())
print("총에너지:", p.total_energy())
print("운동에너지:", p.kinetic_energy())

8. 이상기체 상태 방정식 🌡️

class IdealGas:
    R = 8.314

    def __init__(self, n, temperature, volume):
        self.n = n
        self.temperature = temperature
        self.volume = volume

    def pressure(self):
        return self.n * IdealGas.R * self.temperature / self.volume

    def change_temperature(self, new_temperature):
        self.temperature = new_temperature

gas = IdealGas(n=1.0, temperature=300, volume=0.024)

print("압력:", gas.pressure(), "Pa")
gas.change_temperature(600)
print("온도 상승 후 압력:", gas.pressure(), "Pa")

9. 파동의 중첩과 간섭 🌊

제시 코드

import math

class Wave:
    def __init__(self, amplitude, k, omega, phase=0):
        self.amplitude = amplitude
        self.k = k
        self.omega = omega
        self.phase = phase

    def displacement(self, x, t):
        return self.amplitude * math.sin(self.k * x - self.omega * t + self.phase)

class WaveSystem:
    def __init__(self, waves):
        self.waves = waves

    def total_displacement(self, x, t):
        return sum(w.displacement(x, t) for w in self.waves)

w1 = Wave(1.0, 2.0, 3.0, 0)
w2 = Wave(1.0, 2.0, 3.0, math.pi)

system = WaveSystem([w1, w2])

for x in [0, 0.5, 1.0]:
    print(x, system.total_displacement(x, t=0))

10. 양자역학 1차원 무한 퍼텐셜 우물 입자 🧬

class InfiniteWellParticle:
    H = 6.626e-34

    def __init__(self, mass, length):
        self.mass = mass
        self.length = length

    def energy_level(self, n):
        return n**2 * InfiniteWellParticle.H**2 / (8 * self.mass * self.length**2)

    def transition_energy(self, n1, n2):
        return abs(self.energy_level(n2) - self.energy_level(n1))

electron = InfiniteWellParticle(mass=9.11e-31, length=1e-9)

for n in range(1, 4):
    print(n, "번째 에너지 준위:", electron.energy_level(n))

print("1→2 전이 에너지:", electron.transition_energy(1, 2))

13. 값에 따라 마커 색상 변경

import pandas as pd
import folium

df = pd.DataFrame({
    "지역": ["A", "B", "C", "D"],
    "위도": [37.57, 37.55, 37.53, 37.51],
    "경도": [126.98, 127.00, 127.02, 127.04],
    "위험도": [20, 45, 70, 90]
})

def get_color(value):
    if value >= 80:
        return "red"
    elif value >= 50:
        return "orange"
    return "green"

m = folium.Map(location=[37.54, 127.01], zoom_start=12)

for _, row in df.iterrows():
    folium.Marker(
        [row["위도"], row["경도"]],
        popup=f'{row["지역"]}: {row["위험도"]}',
        icon=folium.Icon(color=get_color(row["위험도"]))
    ).add_to(m)

m.save("13_marker_by_value.html")

32. 간단한 단계 구분도

import pandas as pd
import folium

geojson_data = {
    "type": "FeatureCollection",
    "features": [
        {
            "type": "Feature",
            "properties": {"code": "A", "name": "A구역"},
            "geometry": {
                "type": "Polygon",
                "coordinates": [[
                    [126.95, 37.54], [126.98, 37.54],
                    [126.98, 37.57], [126.95, 37.57],
                    [126.95, 37.54]
                ]]
            }
        },
        {
            "type": "Feature",
            "properties": {"code": "B", "name": "B구역"},
            "geometry": {
                "type": "Polygon",
                "coordinates": [[
                    [126.98, 37.54], [127.01, 37.54],
                    [127.01, 37.57], [126.98, 37.57],
                    [126.98, 37.54]
                ]]
            }
        }
    ]
}

df = pd.DataFrame({
    "code": ["A", "B"],
    "value": [25, 80]
})

m = folium.Map(location=[37.555, 126.98], zoom_start=13)

folium.Choropleth(
    geo_data=geojson_data,
    data=df,
    columns=["code", "value"],
    key_on="feature.properties.code",
    fill_color="YlOrRd",
    legend_name="위험 지수"
).add_to(m)

m.save("32_choropleth.html")

49. CSV 파일 기반 학교 지도

import pandas as pd
import folium
from folium.plugins import MarkerCluster

# schools.csv 열: 학교명, 위도, 경도, 학생수
df = pd.read_csv("schools.csv", encoding="utf-8-sig")

m = folium.Map(
    location=[df["위도"].mean(), df["경도"].mean()],
    zoom_start=11
)

cluster = MarkerCluster(name="학교").add_to(m)

for _, row in df.iterrows():
    popup = (
        f'<b>{row["학교명"]}</b><br>'
        f'학생 수: {row["학생수"]}명'
    )

    folium.CircleMarker(
        location=[row["위도"], row["경도"]],
        radius=max(5, row["학생수"] / 100),
        popup=folium.Popup(popup, max_width=250),
        tooltip=row["학교명"],
        fill=True,
        fill_opacity=0.7
    ).add_to(cluster)

folium.LayerControl().add_to(m)
m.save("49_school_map.html")

50. 종합 대시보드형 지도

import pandas as pd
import folium
from folium.plugins import (
    MarkerCluster,
    HeatMap,
    MiniMap,
    Fullscreen,
    MeasureControl
)

df = pd.DataFrame({
    "장소": ["A지점", "B지점", "C지점", "D지점"],
    "위도": [37.56, 37.57, 37.55, 37.58],
    "경도": [126.97, 126.99, 127.00, 126.98],
    "측정값": [30, 80, 55, 95]
})

m = folium.Map(
    location=[df["위도"].mean(), df["경도"].mean()],
    zoom_start=13,
    tiles=None
)

folium.TileLayer("OpenStreetMap", name="일반 지도").add_to(m)
folium.TileLayer("CartoDB positron", name="밝은 지도").add_to(m)

marker_group = folium.FeatureGroup(name="측정 지점")
heat_group = folium.FeatureGroup(name="열지도")

cluster = MarkerCluster().add_to(marker_group)

for _, row in df.iterrows():
    folium.CircleMarker(
        [row["위도"], row["경도"]],
        radius=row["측정값"] / 10,
        popup=f'{row["장소"]}: {row["측정값"]}',
        tooltip=row["장소"],
        fill=True,
        fill_opacity=0.7
    ).add_to(cluster)

HeatMap(
    df[["위도", "경도", "측정값"]].values.tolist(),
    radius=25
).add_to(heat_group)

marker_group.add_to(m)
heat_group.add_to(m)

MiniMap().add_to(m)
Fullscreen().add_to(m)
MeasureControl().add_to(m)
folium.LayerControl(collapsed=False).add_to(m)

m.fit_bounds(df[["위도", "경도"]].values.tolist())
m.save("50_integrated_map.html")

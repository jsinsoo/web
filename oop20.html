15. 반 평균 이상 학생 찾기

class ClassScore:
    def __init__(self, scores): self.scores=scores
    def above_avg(self):
        avg=sum(self.scores.values())/len(self.scores)
        r=[]
        for n,s in self.scores.items():
            if s>=avg: r.append(n)
        return r

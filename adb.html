제시 코드

class DNASequence:
    def __init__(self, sequence):
        self.sequence = sequence.upper()

    def length(self):
        return len(self.sequence)

    def gc_content(self):
        g = self.sequence.count("G")
        c = self.sequence.count("C")
        return (g + c) / self.length() * 100

    def complement(self):
        table = str.maketrans("ATGC", "TACG")
        return self.sequence.translate(table)

dna = DNASequence("ATGCGCGTTA")

print("길이:", dna.length())
print("GC 함량:", round(dna.gc_content(), 2), "%")
print("상보 서열:", dna.complement())

class Gene:
    codon_table = {
        "AUG": "Met", "UUU": "Phe", "UUC": "Phe",
        "GCU": "Ala", "GCC": "Ala", "UAA": "Stop"
    }

    def __init__(self, dna):
        self.dna = dna.upper()

    def transcribe(self):
        return self.dna.replace("T", "U")

    def translate(self):
        mrna = self.transcribe()
        protein = []
        for i in range(0, len(mrna) - 2, 3):
            codon = mrna[i:i+3]
            amino = Gene.codon_table.get(codon, "?")
            if amino == "Stop":
                break
            protein.append(amino)
        return protein

gene = Gene("ATGGCTTTTTAA")

print("mRNA:", gene.transcribe())
print("아미노산:", gene.translate())

제시 코드

class Enzyme:
    def __init__(self, name, vmax, km):
        self.name = name
        self.vmax = vmax
        self.km = km

    def reaction_rate(self, substrate):
        return self.vmax * substrate / (self.km + substrate)

    def report(self, substrate):
        rate = self.reaction_rate(substrate)
        return f"{self.name}: 기질 농도 {substrate}, 반응 속도 {rate:.3f}"

amylase = Enzyme("Amylase", vmax=100, km=5)

for s in [1, 5, 10, 50]:
    print(amylase.report(s))

class CellCulture:
    def __init__(self, initial_cells, division_time):
        self.cells = initial_cells
        self.division_time = division_time
        self.time = 0

    def step(self, hours):
        self.time += hours
        divisions = int(self.time // self.division_time)
        self.cells = self.cells * (2 ** divisions)
        self.time = self.time % self.division_time

    def count(self):
        return self.cells

culture = CellCulture(initial_cells=100, division_time=2)

for hour in [1, 1, 2, 2]:
    culture.step(hour)
    print("현재 세포 수:", culture.count())

5. 로지스틱 개체군 성장 모델 🐇

제시 코드

class Population:
    def __init__(self, size, growth_rate, capacity):
        self.size = size
        self.r = growth_rate
        self.k = capacity

    def step(self, dt):
        change = self.r * self.size * (1 - self.size / self.k) * dt
        self.size += change

    def report(self):
        return round(self.size, 2)

rabbit = Population(size=50, growth_rate=0.4, capacity=500)

for t in range(10):
    rabbit.step(dt=1)
    print(t + 1, "년 후 개체 수:", rabbit.report())

6. 포식자-피식자 모델 🦊🐰

제시 코드

class PredatorPrey:
    def __init__(self, prey, predator, a, b, c, d):
        self.prey = prey
        self.predator = predator
        self.a = a  # 피식자 자연 증가율
        self.b = b  # 포식에 의한 피식자 감소율
        self.c = c  # 포식자 자연 감소율
        self.d = d  # 먹이 섭취에 의한 포식자 증가율

    def step(self, dt):
        prey_change = (self.a * self.prey - self.b * self.prey * self.predator) * dt
        predator_change = (-self.c * self.predator + self.d * self.prey * self.predator) * dt
        self.prey += prey_change
        self.predator += predator_change

    def state(self):
        return round(self.prey, 2), round(self.predator, 2)

system = PredatorPrey(prey=40, predator=9, a=0.8, b=0.02, c=0.6, d=0.01)

for t in range(8):
    system.step(0.1)
    print("피식자, 포식자:", system.state())

제시 코드

class Neuron:
    def __init__(self, threshold):
        self.threshold = threshold
        self.potential = 0
        self.fired = False

    def receive(self, signal):
        self.potential += signal
        if self.potential >= self.threshold:
            self.fire()

    def fire(self):
        self.fired = True
        print("⚡ 뉴런 발화!")
        self.potential = 0

neuron = Neuron(threshold=10)

for signal in [2, 3, 4, 5, 1]:
    neuron.receive(signal)
    print("현재 막전위:", neuron.potential)

9. 유전 알고리즘으로 적응도 선택 🧬🤖

제시 코드

import random

class Organism:
    def __init__(self, genes):
        self.genes = genes

    def fitness(self):
        return sum(self.genes)

    def mutate(self, rate=0.1):
        self.genes = [
            1 - g if random.random() < rate else g
            for g in self.genes
        ]

    def crossover(self, other):
        point = len(self.genes) // 2
        child_genes = self.genes[:point] + other.genes[point:]
        return Organism(child_genes)

parent1 = Organism([1, 0, 1, 1, 0, 1])
parent2 = Organism([0, 1, 0, 1, 1, 0])

child = parent1.crossover(parent2)
child.mutate(rate=0.2)

print("자식 유전자:", child.genes)
print("적응도:", child.fitness())

10. 감염병 확산 SIR 모델 🦠📈

제시 코드

class SIRModel:
    def __init__(self, susceptible, infected, recovered, beta, gamma):
        self.s = susceptible
        self.i = infected
        self.r = recovered
        self.beta = beta
        self.gamma = gamma

    def step(self, dt):
        population = self.s + self.i + self.r
        new_infected = self.beta * self.s * self.i / population * dt
        new_recovered = self.gamma * self.i * dt

        self.s -= new_infected
        self.i += new_infected - new_recovered
        self.r += new_recovered

    def state(self):
        return round(self.s, 1), round(self.i, 1), round(self.r, 1)

model = SIRModel(susceptible=990, infected=10, recovered=0, beta=0.3, gamma=0.1)

for day in range(10):
    model.step(dt=1)
    print(day + 1, "일:", model.state())
